    public static int findStep(int n) {

        int[] arr = new int[n + 1];

        // base cases
        arr[0] = 1;
        arr[1] = 1;
        arr[2] = 2;
    
        // fibonacci w/o recursion 
        for (int i = 3; i <= n; i++) {
            arr[i] = arr[i - 1] + arr[i - 2];
        }
    
        return arr[n];
        
    }
